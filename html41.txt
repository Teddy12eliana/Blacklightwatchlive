<script>
  async function fetchAlerts() {
    try {
      const response = await fetch("https://live-w8od.onrender.com/status-api");
      if (!response.ok) throw new Error('Network response was not OK');
      const data = await response.json();
      document.getElementById("alertBox").innerText = JSON.stringify(data, null, 2);
    } catch (error) {
      document.getElementById("alertBox").innerText = "‚ùå Failed to fetch alerts.";
      console.error("Alert Fetch Error:", error);
    }
  }

  fetchAlerts();
</script>
